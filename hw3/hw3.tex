\subsection*{1. $L=\{a^p|p\ prime\ number\}$}
proof:\\
fooling set F=a* \\
$\forall x=a^i,y=a^j\in F,i\not=j$, 
assume $i \leq j$ \\
let the prime number $p \geq i$ \\
$p+0(j-i)=p$ is prime \\
$p+p(j-i)\geq p$ is not prime \\
$\therefore \exists integer\ k\in (0,p]\ s.t.$ \\
$p+(k-1)(j-i)$ is prime \\
$p+k(j-i)$ is not prime \\
let $z=a^{p+(k-1)j-ki}$ \\
$xz=a^{p+(k-1)(j-i)}\in L$ \\
$yz=a^{p+k(j-i)}\notin L$ \\
$\therefore$ F is the fooling set of L \\
$\because$ F is infinite \\
$\therefore$ L is not regular

\subsection*{2. $L=\{a^{n^2}|n \geq 1\}$}
proof:\\
fooling set $F=L=\{a^{n^2}|n \geq 1\}$\\
$\forall x=a^{i^2},y=a^{j^2}\in F,i\not=j$, assume $i \leq j$\\
let $z=a^{2i+1}$\\
$xz=a^{i^2+2i+1}=a^{(i+1)^2}\in L$\\
$yz=a^{j^2+2i+1}$\\
the next square number after $j^2$ is $(j+1)^2=j^2+2j+1>j^2+2i+1>j^2$\\
$\therefore j^2+2i+1$ is not a square number \\
$\therefore yz \notin L$\\
$\therefore$ F is the fooling set of L \\
$\because$ F is infinite \\
$\therefore$ L is not regular
